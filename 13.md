var lengthOfLongestSubstring = function(s) {
    let map = new Map();
    let left = 0, maxLength = 0;

    for (let right = 0; right < s.length; right++) {
        if (map.has(s[right])) {
            // –ø—Ä—ã–≥–∞–µ–º —Å—Ä–∞–∑—É –∑–∞ –ø–æ—Å–ª–µ–¥–Ω–∏–π –¥—É–±–ª—å

        map.set(s[right], right);
        maxLength = Math.max(maxLength, right - left + 1);
    }

    return maxLength;
};
